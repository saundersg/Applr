set.seed(121)

num_rows <- 1000 #1460 total
keep <- sample(1:nrow(train), num_rows)

mytrain <- train[keep, ]
mytest <- train[-keep, ]

# Compute R-squared for validation
yhT <- predict(lm5, newdata=mytest)
ybar <- mean(log(mytest$SalePrice)) # Log-transform SalePrice

# Compute SSTO
SSTO <- sum((log(mytest$SalePrice) - ybar)^2)

# Compute SSE
SSEt <- sum((log(mytest$SalePrice) - yhT)^2)

# Compute R-squared
rst <- 1 - SSEt/SSTO

# Compute adjusted R-squared
n <- length(mytest$SalePrice) #sample size
pt <- length(coef(lm5)) #num. parameters in model
rsta <- 1 - (n-1)/(n-pt)*SSEt/SSTO

my_output_table2 <- data.frame(
  Model = "True",
  `Original R2` = summary(lm5)$r.squared,
  `Orig. Adj. R-squared` = summary(lm5)$adj.r.squared,
  `Validation R-squared` = rst,
  `Validation Adj. R^2` = rsta)

colnames(my_output_table2) <- c("Model", "Original $R^2$", "Original Adj. $R^2$", "Validation $R^2$", "Validation Adj. $R^2$")

pander(print(my_output_table2))
