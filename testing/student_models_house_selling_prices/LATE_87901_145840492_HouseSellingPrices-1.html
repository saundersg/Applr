set.seed(123)

num_rows <- 1000
keep <- sample(1:nrow(train), num_rows)

mytrain <- train[keep, ] #Use this in the lm(..., data=mytrain) it is like "rbdata"

mytest <- train[-keep, ] #Use this in the predict(..., newdata=mytest) it is like "rbdata2"
lm_t <- lm(SalePrice ~ TotalSF + RichNbrhd + TotalSF:RichNbrhd * Baths * OverallQual, data=mytrain)

yhtt <- predict(lm_t, newdata=mytest)

ybar <- mean(mytest$SalePrice)

SSTO <- sum((mytest$SalePrice - ybar)^2)

SSE <- sum((mytest$SalePrice - yhtt)^2)

rst <- 1 - SSE/SSTO

n <- length(mytest$SalePrice)
pt <- length(coef(lm_t))

rsta <- 1 - (n-1)/(n-pt)*SSE/SSTO

output_table <- data.frame(Model = 'Validated', `OG R-Squared` = summary(lm_t)$r.squared, `Adjusted OG R-Squared` = summary(lm_t)$adj.r.squared, `Validated R-Squared` = rst, `Validated Adj. R-Squared`=rsta)

colnames(output_table) <- c('Model','Original $R^2$', 'Original Adj. $R^2$', 'Validation $R^2$','Validation Adj. $R^2$')

knitr::kable(output_table, escape=T, digits=4)
